```go
// internal/config/config.go

type Config struct {
    Server        ServerConfig        `yaml:"server"`
    Database      DatabaseConfig      `yaml:"database"`
    Logging       LoggingConfig       `yaml:"logging"`
    FFmpeg        FFmpegConfig        `yaml:"ffmpeg"`
    Transcoding   TranscodingConfig   `yaml:"transcoding"`
    Library       LibraryConfig       `yaml:"library"`
    DLNA          DLNAConfig          `yaml:"dlna"`
    WebSocket     WebSocketConfig     `yaml:"websocket"`
}

func Load(path string) (*Config, error) {
    data, err := os.ReadFile(path)
    if err != nil {
        return nil, fmt.Errorf("reading config: %w", err)
    }

    var cfg Config
    if err := yaml.Unmarshal(data, &cfg); err != nil {
        return nil, fmt.Errorf("parsing config: %w", err)
    }

    return &cfg, nil
}
```

### 4.3 Item Repository

```go
// internal/data/repositories/item.go

type ItemRepository struct {
    *BaseRepository
}

func (r *ItemRepository) GetByID(ctx context.Context, id string) (*models.Item, error) {
    sql := `SELECT Id, Name, Type, Path, ParentId, PremiereDate,
                   ProductionYear, SortName, Overview, CommunityRating,
                   RunTimeTicks, PrimaryImageId
            FROM TypedBaseItems WHERE Id = ?`

    var item models.Item
    row := r.db.QueryRow(sql, id)
    err := row.Scan(...)
    if err != nil {
        return nil, err
    }
    return &item, nil
}

func (r *ItemRepository) GetChildren(ctx context.Context, parentID string) ([]*models.Item, error) {
    sql := `SELECT Id, Name, Type, Path, ParentId, PremiereDate,
                   ProductionYear, SortName, Overview, CommunityRating,
                   RunTimeTicks, PrimaryImageId
            FROM TypedBaseItems WHERE ParentId = ?`

    rows, err := r.db.Query(sql, parentID)
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    var items []*models.Item
    for rows.Next() {
        var item models.Item
        err := rows.Scan(...)
        if err != nil {
            return nil, err
        }
        items = append(items, &item)
    }
    return items, nil
}
```

## 5. Authentication

### 5.1 Password Hashing

```go
// internal/auth/password.go

import (
    "crypto/pbkdf2"
    "encoding/hex"
)

const (
    PBKDF2Iterations = 100000
    SaltLength       = 4
    HashLength       = 4
)

func HashPassword(password, saltHex string) string {
    salt, _ := hex.DecodeString(saltHex)
    hash := pbkdf2.Key([]byte(password), salt, PBKDF2Iterations, HashLength, sha256.New)
    return hex.EncodeToString(hash)
}

func VerifyPassword(password, saltHex, expectedHashHex string) bool {
    computedHash := HashPassword(password, saltHex)
    return computedHash == expectedHashHex
}

func GenerateSalt() string {
    salt := make([]byte, SaltLength)
    rand.Read(salt)
    return hex.EncodeToString(salt)
}
```

### 5.2 Session Authentication

```go
// internal/auth/session.go

type SessionManager struct {
    sessions map[string]*Session
    mu      sync.RWMutex
}

func (sm *SessionManager) Create(userID, deviceID, deviceName string) (*Session, error) {
    token := generateToken() // 32-byte random hex

    session := &Session{
        ID:          generateUUID(),
        UserID:      userID,
        DeviceID:    deviceID,
        DeviceName:  deviceName,
        AccessToken: token,
        Created:     time.Now(),
    }

    sm.mu.Lock()
    sm.sessions[token] = session
    sm.mu.Unlock()

    return session, nil
}

func (sm *SessionManager) Validate(token string) (*Session, error) {
    sm.mu.RLock()
    defer sm.mu.RUnlock()

    session, ok := sm.sessions[token]
    if !ok {
        return nil, ErrInvalidToken
    }
    return session, nil
}
```

## 6. API Handlers

### 6.1 Handler Pattern

```go
// internal/api/handlers/library.go

type LibraryHandler struct {
    itemRepo  *repositories.ItemRepository
    libraryMgr *services.LibraryManager
}

func NewLibraryHandler(itemRepo *repositories.ItemRepository, libraryMgr *services.LibraryManager) *LibraryHandler {
    return &LibraryHandler{
        itemRepo:   itemRepo,
        libraryMgr: libraryMgr,
    }
}

func (h *LibraryHandler) GetItems(w http.ResponseWriter, r *http.Request) {
    ctx := r.Context()

    parentID := chi.URLParam(r, "parentId")
    startIndex, _ := strconv.Atoi(r.URL.Query().Get("startIndex"))
    limit, _ := strconv.Atoi(r.URL.Query().Get("limit"))

    items, total, err := h.itemRepo.GetChildren(ctx, parentID, startIndex, limit)
    if err != nil {
        http.Error(w, err.Error(), 500)
        return
    }

    writeJSON(w, &ItemsResponse{
        Items:           items,
        TotalRecordCount: total,
        StartIndex:      startIndex,
    })
}

func (h *LibraryHandler) GetItem(w http.ResponseWriter, r *http.Request) {
    ctx := r.Context()
    id := chi.URLParam(r, "id")

    item, err := h.itemRepo.GetByID(ctx, id)
    if err != nil {
        if err == sql.ErrNotFound {
            http.Error(w, "Item not found", 404)
            return
        }
        http.Error(w, err.Error(), 500)
        return
    }

    writeJSON(w, item)
}
```

### 6.2 Middleware Chain

```go
// internal/api/middleware.go

func MiddlewareChain(handler http.Handler) http.Handler {
    return middleware.Compose(
        recoveryMiddleware,    // Panic recovery
        loggingMiddleware,      // Request logging
        corsMiddleware,        // CORS headers
        authMiddleware,        // Authentication
        rateLimitMiddleware,   // Rate limiting (future)
    )
}

func recoveryMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        defer func() {
            if err := recover(); err != nil {
                log.Error("panic recovered", "error", err)
                http.Error(w, "Internal Server Error", 500)
            }
        }()
        next.ServeHTTP(w, r)
    })
}

func authMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        token := r.Header.Get("X-Emby-Token")
        if token == "" {
            token = r.URL.Query().Get("api_key")
        }

        if token != "" {
            session, err := sessionManager.Validate(token)
            if err == nil {
                ctx := context.WithValue(r.Context(), "session", session)
                r = r.WithContext(ctx)
            }
        }

        next.ServeHTTP(w, r)
    })
}
```

```go
func (lm *LibraryManager) ScanLibrary(ctx context.Context) error {

    if err != nil {
        return err
    }

    lm.mu.Lock()
    lm.items = items
    lm.mu.Unlock()

    return nil
}

type MediaScanner struct {
    itemRepo *repositories.ItemRepository
}

func (s *MediaScanner) ScanPaths(ctx context.Context, paths []string) ([]*models.Item, error) {
    var items []*models.Item

    for _, path := range paths {
        scanned, err := s.scanDirectory(ctx, path)
        if err != nil {
            log.Warn("scan error", "path", path, "error", err)
            continue
        }
        items = append(items, scanned...)
    }

    return items, nil
}
```

```go
// internal/services/session/manager.go

type SessionManager struct {
    sessions  map[string]*Session
    hub       *websocket.Hub
    mu        sync.RWMutex

func (sm *SessionManager) CreateSession(userID, deviceID, deviceName string) (*Session, error) {
    sm.mu.Lock()

    token := generateToken()
    session := &Session{

    return session, nil

func (sm *SessionManager) UpdatePlayState(token string, state *models.PlayState) error {
    sm.mu.Lock()
    defer sm.mu.Unlock()

    session, ok := sm.sessions[token]
    if !ok {
        return ErrSessionNotFound
    }

    session.PlayState = state
    sm.hub.Broadcast(session)
    return nil
}
```

## 10. Logging

### 10.1 Structured Logging

```go
// internal/logging/logging.go

import "log/slog"

var logger *slog.Logger

func Init(level string, output string) error {
    var handler slog.Handler
    if output == "stdout" {
        handler = slog.NewJSONHandler(os.Stdout, nil)
    } else {
        f, err := os.OpenFile(output, os.O_CREATE|os.O_APPEND|os.O_WRONLY, 0644)
        if err != nil {
            return err
        }
        handler = slog.NewJSONHandler(f, nil)
    }

    var lvl slog.Level
    switch level {
    case "debug":
        lvl = slog.LevelDebug
    case "info":
        lvl = slog.LevelInfo
    case "warn":
        lvl = slog.LevelWarn
    case "error":
        lvl = slog.LevelError
    default:
        lvl = slog.LevelInfo
    }

    logger = slog.New(handler)
    logger = logger.With("service", "emby")
    slog.SetDefault(logger)

    return nil
}
```
