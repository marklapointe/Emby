## 1. Testing Philosophy

### 1.1 Core Principles

1. **Test behavior, not implementation** — Tests should verify functionality, not internal details
2. **Deterministic tests** — All tests must produce consistent results
3. **Fast feedback** — Unit tests should run in milliseconds
4. **Isolation** — Tests must not depend on each other
5. **Real database** — Use actual SQLite for repository tests (not mocks)

### 1.2 Test Pyramid

```
        /\
       /  \
      / E2E\      <- Few, slow, comprehensive
     /------\
    /Integr. \    <- Medium count, medium speed
   /----------\
  /  Unit Tests \ <- Many, fast, isolated
 /--------------\
```

## 2. Test Types

### 2.1 Unit Tests

| Category | Description | Examples |
|----------|-------------|----------|
| Repository | Data access with real SQLite | `ItemRepo.GetByID()`, `UserRepo.Authenticate()` |
| Service | Business logic | `LibraryManager.Scan()`, `SessionManager.Create()` |
| Handler | HTTP behavior | Request parsing, response format |
| Utility | Helper functions | Password hashing, token generation |

### 2.2 Integration Tests

| Category | Description | Examples |
|----------|-------------|----------|
| API | Full HTTP request/response | `GET /emby/Items`, `POST /emby/Users/{id}/Authenticate` |
| Database | Multi-table operations | Transaction rollback, cascade delete |
| Service | Service collaboration | Library scan triggers metadata refresh |

### 2.3 End-to-End Tests

| Category | Description | Examples |
|----------|-------------|----------|
| Client Simulation | Simulate Emby client | Browse library, play media |
| Authentication Flow | Full login process | Username/password → session token |
| Playback Flow | Stream video | Browse → select → stream → transcode |

### 2.4 Performance Tests

| Category | Description | Examples |
|----------|-------------|----------|
| Benchmark | Measure operation time | Repository queries, encoding |
| Load | Concurrent requests | Multiple clients streaming |
| Memory | Memory usage tracking | Long-running server memory |

```go
func setupTestDB(t *testing.T) *sqlite.Conn {
    // Create temp database
    tmpFile := filepath.Join(t.TempDir(), "test.db")

    // Copy schema from reference
    src, err := os.Open("testdata/schema.sql")
    if err != nil {
        t.Fatal(err)
    }
    defer src.Close()

    dst, err := os.Create(tmpFile)
    if err != nil {
        t.Fatal(err)
    }
    defer dst.Close()

    if _, err := io.Copy(dst, src); err != nil {
        t.Fatal(err)
    }

    // Seed test data
    seedTestData(t, tmpFile)

    db, err := sqlite.Open(tmpFile, sqlite.OpenReadWrite)
    if err != nil {
        t.Fatal(err)
    }

    return db
}
```

### 5.1 E2E Test Framework

```go
func TestE2E_PlaybackFlow(t *testing.T) {
    if testing.Short() {
        t.Skip("skipping E2E test")
    }

    // Start server
    server := startTestServer()
    defer server.Shutdown()

    // Create client
    client := NewEmbyClient(server.URL)

    // Login
    session, err := client.Login("testuser", "testpass")
    if err != nil {
        t.Fatalf("login failed: %v", err)
    }

    // Browse library
    items, err := client.GetItems(session.Token, "library-1")
    if err != nil {
        t.Fatalf("browse failed: %v", err)
    }

    if len(items) == 0 {
        t.Fatal("no items in library")
    }

    // Get stream URL
    streamURL, err := client.GetStreamURL(session.Token, items[0].ID)
    if err != nil {
        t.Fatalf("get stream failed: %v", err)
    }

    // Verify stream is accessible
    resp, err := http.Head(streamURL)
    if err != nil {
        t.Fatalf("stream not accessible: %v", err)
    }
    resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        t.Errorf("stream status %d", resp.StatusCode)
    }
}
```

```go
        t.Skip("skipping E2E test")
    }

    server := startTestServer()
    defer server.Shutdown()

    // Connect WebSocket

    conn, _, err := websocket.DefaultDialer.Dial(url, nil)
    if err != nil {
        t.Fatalf("WebSocket connect failed: %v", err)
    }
    defer conn.Close()

    // Send login
    loginMsg := WebSocketMessage{
        MessageType: "Login",
        Data: map[string]string{
            "Username": "testuser",
            "Pw":      "testpass",
        },
    }
    if err := conn.WriteJSON(loginMsg); err != nil {
        t.Fatalf("login send failed: %v", err)
    }

    // Read login result
    var result LoginResultMessage
    if err := conn.ReadJSON(&result); err != nil {
        t.Fatalf("login response failed: %v", err)
    }

    if !result.Success {
        t.Error("login failed")
    }
}
```

```go
func TestLoad_ConcurrentStreaming(t *testing.T) {
    if testing.Short() {
        t.Skip("skipping load test")
    }

    server := startTestServer()

    const numClients = 10
    const duration = 30 * time.Second

    var wg sync.WaitGroup
    errorCh := make(chan error, numClients)

    for i := 0; i < numClients; i++ {
        wg.Add(1)
        go func(clientID int) {
            defer wg.Done()

            client := NewEmbyClient(server.URL)
            start := time.Now()

            for time.Since(start) < duration {
                _, err := client.GetStreamURL("test-token", "movie-1")
                if err != nil {
                    errorCh <- err
                    return
                }
            }
        }(i)
    }

    wg.Wait()
    close(errorCh)

    // Check for errors
    for err := range errorCh {
        t.Errorf("client error: %v", err)
    }
}
```

## 8. Test Harnesses

### 8.1 Test Database Harness

```go
// tests/integration/harness.go

type TestHarness struct {
    DB     *sqlite.Conn
    Server *httptest.Server
    Config *config.Config
    Client *EmbyClient
}

func NewTestHarness(t *testing.T) *TestHarness {
    // Create temporary database
    tmpDir := t.TempDir()
    dbPath := filepath.Join(tmpDir, "test.db")

    // Copy schema
    copyFile(t, "testdata/schema.sql", dbPath)

    // Seed data
    copyFile(t, "testdata/seed.sql", dbPath)

    db, err := sqlite.Open(dbPath, sqlite.OpenReadWrite)
    if err != nil {
        t.Fatal(err)
    }

    // Create server
    cfg := &config.Config{
        Database: config.DatabaseConfig{Path: dbPath},
        Server:   config.ServerConfig{Port: 0}, // Pick available port
    }

    server := httptest.NewServer(createTestRouter(cfg))

    return &TestHarness{
        DB:     db,
        Server: server,
        Config: cfg,
        Client: NewEmbyClient(server.URL),
    }
}

func (h *TestHarness) Close() {
    h.DB.Close()
    h.Server.Close()
}
```

## 9. Safety Rules

### 9.1 Test Environment Rules

1. **Never connect to production database** — Use isolated test database only
2. **Mock external services** — FFmpeg, network requests
3. **Cleanup after tests** — Remove temp files, close connections
4. **Deterministic data** — Use fixed seeds for random data

### 9.2 CI/CD Rules

1. **Run unit tests on every commit** — Must pass
2. **Run integration tests on PR** — Must pass before merge
3. **Run E2E tests nightly** — Can fail without blocking merge
4. **Performance tests weekly** — Track trends

### 9.3 Test Data Rules

1. **No real user data** — Use synthetic test accounts
2. **No secrets in tests** — Use test credentials only
3. **Minimal test data** — Only data needed for test
4. **Regenerate seeds** — Update seed data when schema changes
